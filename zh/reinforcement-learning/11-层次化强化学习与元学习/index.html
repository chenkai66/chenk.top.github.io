import numpy as np
from collections import defaultdict

class IntraOptionQLearning:
    """зҰ»ж•ЈйҖүйЎ№йӣҶеҗҲдёҠзҡ„ intra-option Q-learningгҖӮ

    жҜҸдёҖжқЎеҺҹе§ӢиҪ¬з§»йғҪдјҡжӣҙж–°пјҡ
    (1) еҪ“еүҚйҖүйЎ№зҡ„й«ҳеұӮ Q(s, o)пјӣ
    (2) иҜҘйҖүйЎ№зҡ„еҶ…йғЁ Q иЎЁпјҢд»ҺиҖҢйҡҗејҸдҝ®жӯЈе…¶еҶ…йғЁзӯ–з•ҘгҖӮ
    """

    def __init__(self, env, options, alpha=0.5, gamma=0.99, epsilon=0.1):
        self.env, self.options = env, options
        self.alpha, self.gamma, self.epsilon = alpha, gamma, epsilon
        self.Q = defaultdict(lambda: np.zeros(len(options)))     # Q(s, o)
        self.Q_prim = defaultdict(lambda: np.zeros(env.n_actions))

    def _select_option(self, state):
        available = [i for i, o in enumerate(self.options)
                     if o.can_initiate(state)]
        if not available:
            return None
        if np.random.rand() < self.epsilon:
            return int(np.random.choice(available))
        q = self.Q[state][available]
        return int(available[np.argmax(q)])

    def train_episode(self, max_steps=1000):
        state = self.env.reset()
        total_reward = 0.0
        for _ in range(max_steps):
            oid = self._select_option(state)
            if oid is None:
                break
            option = self.options[oid]

            while not option.should_terminate(state):
                action = option.act(state, self.epsilon)
                next_state, reward, done = self.env.step(action)
                total_reward += reward

                # и®Ўз®—е»¶з»ӯеҖј U(s', o)
                if option.should_terminate(next_state):
                    U = np.max(self.Q[next_state])
                else:
                    U = self.Q[next_state][oid]

                # жӣҙж–°й«ҳеұӮ Q еҖј
                self.Q[state][oid] += self.alpha * (
                    reward + self.gamma * U - self.Q[state][oid])

                # жӣҙж–°еҶ…йғЁзӯ–з•ҘпјҲеҹәдәҺеҺҹе§ӢеҠЁдҪңзҡ„жүҒе№і Q иЎЁпјү
                td = (reward + self.gamma
                      * np.max(self.Q_prim[next_state])
                      - self.Q_prim[state][action])
                self.Q_prim[state][action] += self.alpha * td
                option.policy[state] = int(np.argmax(self.Q_prim[state]))

                state = next_state
                if done:
                    return total_reward
        return total_reward

еңЁз»Ҹе…ёзҡ„ Four Rooms еҹәеҮҶжөӢиҜ•дёӯпјҢдҪҝз”ЁеӣӣдёӘжүӢе·Ҙи®ҫи®Ўзҡ„вҖңз©ҝиҝҮй—ЁеҸЈвҖқйҖүйЎ№еҗҺпјҢintra-option Q-learning зҡ„ж”¶ж•ӣйҖҹеәҰжҜ”жүҒе№і Q-learning еҝ« 3вҖ“5 еҖҚпјҢдё”еӯҰеҲ°зҡ„йҖүйЎ№иғҪзӣҙжҺҘиҝҒз§»еҲ°ж–°зӣ®ж ҮдҪҚзҪ®гҖӮ

MAXQпјҡжІҝд»»еҠЎж ‘еҲҶи§ЈеҖјеҮҪж•°#

$$Q_i(s, a) = V_a(s) + C_i(s, a),$$

е…¶дёӯ $V_a(s)$ жҳҜе®ҢжҲҗеӯҗд»»еҠЎ $a$ зҡ„д»·еҖјпјҢ$C_i(s, a)$ жҳҜе®ҢжҲҗеҮҪж•°вҖ”вҖ”еҚіеӯҗд»»еҠЎз»“жқҹеҗҺз»§з»ӯе®ҢжҲҗзҲ¶д»»еҠЎ $i$ жүҖеёҰжқҘзҡ„йўқеӨ–д»·еҖјгҖӮз”ұдәҺ $V_a$ д»…дҫқиө–дәҺ $a$ пјҢжүҖжңүи°ғз”Ё $a$ зҡ„зҲ¶д»»еҠЎеқҮеҸҜеӨҚз”ЁеҗҢдёҖд»Ҫ $V_a$ пјҢиҝҷжӯЈжҳҜж ·жң¬ж•ҲзҺҮжҸҗеҚҮзҡ„жқҘжәҗгҖӮ

дҪҶ MAXQ зҡ„д»Јд»·жҳҜеҸӘдҝқиҜҒйҖ’еҪ’жңҖдјҳжҖ§иҖҢйқһе…ЁеұҖжңҖдјҳжҖ§пјҡе…¶зӯ–з•ҘжҳҜеңЁз»ҷе®ҡд»»еҠЎеҲҶи§ЈдёӢзҡ„жңҖдјҳи§ЈгҖӮиӢҘд»»еҠЎеӣҫж— жі•иЎЁиҫҫзңҹжӯЈзҡ„жңҖдјҳзӯ–з•ҘпјҢMAXQ дҫҝж— жі•жүҫеҲ°е®ғгҖӮ

зҰ»ж•ЈйҖүйЎ№йҡҫд»Ҙжү©еұ•пјҡеңЁиҝһз»ӯжҺ§еҲ¶жҲ–еғҸзҙ иҫ“е…ҘеңәжҷҜдёӯпјҢжҲ‘д»¬ж— жі•жүӢе·ҘжһҡдёҫеҮәжңүз”Ёзҡ„йҖүйЎ№йӣҶеҗҲгҖӮFeudal NetworksпјҲFuNпјҢVezhnevets зӯүпјҢ2017пјүе’Ң HIROпјҲNachum зӯүпјҢ2018пјүжҸҗеҮәз”Ёй«ҳеұӮ Manager иҫ“еҮәзҡ„иҝһз»ӯзӣ®ж Үеҗ‘йҮҸеҸ–д»ЈзҰ»ж•ЈйҖүйЎ№гҖӮ

$$r^{\text{int}}_t = \cos\!\big(s_{t+c} - s_t,\, g_t\big),$$

иҖҢ Manager еҲҷеҹәдәҺзҺҜеўғзҡ„еӨ–йғЁеҘ–еҠұиҝӣиЎҢи®ӯз»ғгҖӮиҝҷз§Қи§ЈиҖҰи®ҫи®ЎжӯЈжҳҜ Feudal жһ¶жһ„зҡ„дјҳеҠҝжүҖеңЁпјҡWorker еңЁзЁ еҜҶгҖҒеҮ дҪ•еҢ–зҡ„еҶ…еңЁеҘ–еҠұдёҠеӯҰд№ иҝҗеҠЁжҺ§еҲ¶пјҢManager еҲҷдё“жіЁдәҺй•ҝж—¶зЁӢдҝЎз”ЁеҲҶй…ҚпјҢжңүж•Ҳж—¶еәҸй•ҝеәҰзј©зҹӯдёә $T/c$ гҖӮ

HIRO зҡ„еӯҗзӣ®ж ҮйҮҚж ҮжіЁжҠҖе·§#

$$\tilde g_t = \arg\max_{g} \log \pi_\phi(a_{t:t+c} \mid s_{t:t+c},\, g).$$

жӯӨдёҫзЎ®дҝқ Worker зҡ„и®ӯз»ғж•°жҚ®е§Ӣз»ҲдёҺе…¶еҪ“еүҚеҸӮж•°дҝқжҢҒдёҖиҮҙпјҢжҳҫи‘—жҸҗеҚҮдәҶ Manager зҡ„зҰ»зәҝеӯҰд№ зЁіе®ҡжҖ§гҖӮ

дёүгҖҒзӣ®ж ҮжқЎд»¶ RL дёҺ HER#

зӣ®ж ҮжқЎд»¶зӯ–з•Ҙ $\pi(a \mid s, g)$ еҖјеҫ—еҚ•зӢ¬и®Ёи®әпјҢеӣ дёәе®ғжҳҜиҝһжҺҘеұӮж¬ЎеҢ–дёҺе…ғеӯҰд№ зҡ„жЎҘжўҒпјҡеҗҢдёҖзҪ‘з»ңеҸҜиҝҪжұӮеӨҡз§Қзӣ®ж ҮпјҢиҖҢзӣ®ж Ү $g$ д»…жҳҜдёҖдёӘйўқеӨ–иҫ“е…ҘгҖӮ

з»Ҹе…ёеҪўејҸжҳҜ йҖҡз”ЁеҖјеҮҪж•°йҖјиҝ‘еҷЁпјҲUVFAпјҢSchaul зӯүпјҢ2015пјүпјҡеӯҰд№  $V(s, g)$ жҲ– $Q(s, a, g)$ иҖҢйқһдј з»ҹзҡ„ $V(s)$ гҖҒ$Q(s, a)$ гҖӮдҪҶиӢҘж— йўқеӨ–жҠҖе·§пјҢUVFA еңЁзЁҖз–ҸеҘ–еҠұдёӢиЎЁзҺ°жһҒе·®вҖ”вҖ”еӨ§еӨҡж•°зӣ®ж ҮеңЁжҺўзҙўдёӯд»ҺжңӘиҫҫжҲҗпјҢеҘ–еҠұдҝЎеҸ·еҮ д№Һдёәйӣ¶гҖӮ

$$(s_t, a_t, r, s_{t+1}, g) \;\longrightarrow\; (s_t, a_t, r', s_{t+1},\, g' = s_T).$$

з»“еҗҲзҰ»зәҝзӯ–з•Ҙз®—жі•пјҲDDPGгҖҒSACпјүпјҢHER е°ҶзЁҖз–ҸеҘ–еҠұдёӢзҡ„зӣ®ж ҮиҫҫжҲҗд»»еҠЎд»ҺвҖңеҮ д№ҺдёҚеҸҜиғҪвҖқеҸҳдёәвҖңеёёи§„ж“ҚдҪңвҖқгҖӮ

еӣӣгҖҒMeta-RLпјҡеӯҰдјҡеӯҰд№ #

Meta-RL дёҚеҶҚеҒҮи®ҫеҚ•дёҖ MDPпјҢиҖҢжҳҜиҖғиҷ‘дёҖдёӘд»»еҠЎеҲҶеёғ $p(\mathcal{T})$ гҖӮе…ғи®ӯз»ғйҳ¶ж®өпјҢжҷәиғҪдҪ“жҺҘи§ҰеӨ§йҮҸд»»еҠЎ $\mathcal{T}_i \sim p(\mathcal{T})$ пјӣе…ғжөӢиҜ•йҳ¶ж®өпјҢйқўеҜ№ж–°д»»еҠЎж—¶йңҖд»ҘжңҖе°‘дәӨдә’е®ҢжҲҗйҖӮеә”гҖӮ

дё»жөҒж–№жі•еҲҶдёәдёӨзұ»пјҡ

  • еҹәдәҺдјҳеҢ–пјҲMAMLгҖҒReptileгҖҒANILпјүпјҡйҖӮеә” = еңЁжөӢиҜ•д»»еҠЎдёҠжү§иЎҢеҮ жӯҘжўҜеәҰжӣҙж–°гҖӮ
  • еҹәдәҺдёҠдёӢж–Ү / еҫӘзҺҜпјҲRLВІгҖҒPEARLпјүпјҡйҖӮеә” = ж•°жҚ®жөҒе…Ҙж—¶жӣҙж–° RNN йҡҗи—ҸзҠ¶жҖҒпјҢжөӢиҜ•ж—¶ж— йңҖжўҜеәҰи®Ўз®—гҖӮ

MAML вҖ”вҖ” еӯҰд№ дёҖдёӘиүҜеҘҪзҡ„еҲқе§ӢеҢ–#

$$ \theta_i' = \theta - \alpha \nabla_\theta \mathcal{L}_{\mathcal{T}_i}(\theta), \qquad \theta \leftarrow \theta - \beta \nabla_\theta \sum_{i} \mathcal{L}_{\mathcal{T}_i}(\theta_i'). $$

еӨ–еұӮжўҜеәҰйңҖз©ҝиҝҮеҶ…еұӮжӣҙж–°еҸҚеҗ‘дј ж’ӯпјҢеӣ жӯӨеҢ…еҗ« Hessian йЎ№ $\nabla^2 \mathcal{L}$ пјҢиҝҷд№ҹжҳҜ MAML и®Ўз®—жҳӮиҙөзҡ„еҺҹеӣ гҖӮFOMAML зӣҙжҺҘеҝҪз•ҘдәҢйҳ¶йЎ№пјҢе®һиҜҒиЎЁжҳҺе…¶жҜҸж¬ЎеӨ–еұӮжӣҙж–°еҝ«зәҰ $10\times$ пјҢжңҖз»ҲеӣһжҠҘжҚҹеӨұдёҚеҲ° 5%гҖӮ

е·ҰеӣҫжҸҗдҫӣдәҶжңҖзӣҙи§Ӯзҡ„зҗҶи§Јпјҡе…ғи®ӯз»ғе№¶йқһеҜ»жүҫеҜ№еҚ•дёӘд»»еҠЎжңҖдјҳзҡ„еҲқе§ӢеҢ–пјҢиҖҢжҳҜеҜ»жүҫдёҖдёӘвҖңз”ңеҢәвҖқвҖ”вҖ”д»ҺиҝҷйҮҢеҮәеҸ‘пјҢдёҖжӯҘжўҜеәҰеҚіеҸҜжҺҘиҝ‘еҗ„д»»еҠЎзҡ„зү№е®ҡжңҖдјҳи§ЈгҖӮ

import torch
import torch.nn as nn

class GaussianPolicy(nn.Module):
    """иҝһз»ӯжҺ§еҲ¶дёӢзҡ„еҜ№и§’й«ҳж–Ҝзӯ–з•Ҙ"""

    def __init__(self, state_dim=2, action_dim=2, hidden=64):
        super().__init__()
        self.trunk = nn.Sequential(
            nn.Linear(state_dim, hidden), nn.ReLU(),
            nn.Linear(hidden, hidden), nn.ReLU(),
        )
        self.mean = nn.Linear(hidden, action_dim)
        self.log_std = nn.Parameter(torch.zeros(action_dim))

    def sample(self, state):
        h = self.trunk(state)
        mean = torch.tanh(self.mean(h))
        std = self.log_std.exp().clamp(1e-3, 1.0)
        dist = torch.distributions.Normal(mean, std)
        a = dist.rsample()
        return a, dist.log_prob(a).sum(-1)

class MAML:
    """REINFORCE йЈҺж јзӯ–з•ҘжўҜеәҰзҡ„дёҖйҳ¶жҲ–дәҢйҳ¶ MAML"""

    def __init__(self, policy, inner_lr=0.1, outer_lr=1e-3,
                 first_order=False):
        self.policy = policy
        self.inner_lr = inner_lr
        self.first_order = first_order
        self.opt = torch.optim.Adam(policy.parameters(), lr=outer_lr)

    def _rollout(self, env, max_steps=50):
        state = env.reset()
        rewards, log_probs = [], []
        for _ in range(max_steps):
            s = torch.as_tensor(state, dtype=torch.float32).unsqueeze(0)
            a, lp = self.policy.sample(s)
            state, r, done = env.step(a.detach().numpy()[0])
            rewards.append(r)
            log_probs.append(lp)
            if done:
                break
        return rewards, log_probs

    @staticmethod
    def _returns(rewards, gamma=0.99):
        G, out = 0.0, []
        for r in reversed(rewards):
            G = r + gamma * G
            out.insert(0, G)
        out = torch.as_tensor(out, dtype=torch.float32)
        return (out - out.mean()) / (out.std() + 1e-8)

    def _loss(self, rewards, log_probs):
        returns = self._returns(rewards)
        return -(torch.stack(log_probs) * returns).mean()

    def adapt(self, env):
        """еңЁ env дёҠеҒҡдёҖжӯҘеҶ…еұӮжӣҙж–°пјҢиҝ”еӣһйҖӮеә”еҗҺзҡ„еҸӮж•°еҲ—иЎЁ"""
        rewards, log_probs = self._rollout(env)
        loss = self._loss(rewards, log_probs)
        grads = torch.autograd.grad(
            loss, self.policy.parameters(),
            create_graph=not self.first_order,
        )
        return [p - self.inner_lr * g
                for p, g in zip(self.policy.parameters(), grads)]

    def meta_step(self, task_envs):
        meta_loss = 0.0
        for env in task_envs:
            adapted = self.adapt(env)
            # жҡӮеӯҳеҺҹеҸӮж•°пјҢе°ҶйҖӮеә”еҗҺзҡ„еҸӮж•°иЈ…е…ҘзҪ‘з»ңиҝӣиЎҢиҜ„дј°еүҚеҗ‘
            backup = [p.detach().clone() for p in self.policy.parameters()]
            for p, a in zip(self.policy.parameters(), adapted):
                p.data = a.data
            rewards, log_probs = self._rollout(env)
            meta_loss += self._loss(rewards, log_probs)
            for p, b in zip(self.policy.parameters(), backup):
                p.data = b
        meta_loss /= len(task_envs)

        self.opt.zero_grad()
        meta_loss.backward()
        self.opt.step()
        return meta_loss.item()

дёҠиҝ°е®һзҺ°йҖҡиҝҮжӣҝжҚўжқғйҮҚеҗҺжү§иЎҢеүҚеҗ‘дј ж’ӯе®ҢжҲҗе…ғиҜ„дј°пјҢиҖҢйқһдҪҝз”ЁеҮҪж•°ејҸеүҚеҗ‘дј ж’ӯгҖӮиҝҷжҳҜжңҖз®ҖжҙҒжҳ“иҜ»зҡ„зүҲжң¬пјҢеҜ№е°ҸеһӢзҪ‘з»ңе·Іи¶іеӨҹгҖӮиӢҘз”ЁдәҺз ”з©¶зә§ MAMLпјҢе»әи®®дҪҝз”Ё higher жҲ– torch.func.functional_call зӯүеҮҪж•°ејҸ APIгҖӮ

RLВІ вҖ”вҖ” е°Ҷз®—жі•жҠҳеҸ иҝӣ RNN#

$$x_t = (s_t,\, a_{t-1},\, r_{t-1},\, d_{t-1}),$$

еҚізҠ¶жҖҒеҠ дёҠеүҚдёҖжӯҘзҡ„еҠЁдҪңгҖҒеҘ–еҠұе’Ңз»Ҳжӯўж Үеҝ—гҖӮеңЁе…ғиҜ•йӘҢзҡ„еӨҡдёӘеӣһеҗҲдёӯпјҢйҡҗи—ҸзҠ¶жҖҒ $h_t$ дёҚж–ӯзҙҜз§ҜеҪ“еүҚд»»еҠЎдҝЎжҒҜвҖ”вҖ”жң¬иҙЁдёҠйҡҗејҸжү§иЎҢиҙқеҸ¶ж–ҜдҝЎеҝөжӣҙж–°гҖӮе…ғи®ӯз»ғж—¶пјҢеӨ–еұӮдјҳеҢ–еҷЁпјҲPPO жҲ– A2Cпјүи°ғж•ҙ RNN жқғйҮҚпјҢдҪҝиҝҷеҘ—вҖңеҶ…зҪ® RL з®—жі•вҖқеңЁ $p(\mathcal{T})$ дёҠе…·еӨҮй«ҳж•Ҳж ·жң¬еҲ©з”ЁзҺҮгҖӮ

RLВІ жңүдёӨеӨ§дјҳеҠҝпјҡ(i) жөӢиҜ•ж—¶ж— йңҖжўҜеәҰи®Ўз®—пјҢжҜҸжӯҘйҖӮеә”д»…йңҖдёҖж¬ЎеүҚеҗ‘дј ж’ӯпјӣ(ii) йҖӮеә”иҝҮзЁӢжҳҜеӯҰеҮәжқҘзҡ„пјҢеҺҹеҲҷдёҠеҸҜеңЁи®ӯз»ғеҲҶеёғдёҠи¶…и¶Ҡд»»дҪ•жүӢе·Ҙи®ҫи®Ўзҡ„дјҳеҢ–еҷЁгҖӮд»Јд»·жҳҜе…ғи®ӯз»ғж—¶дҝЎз”ЁеҲҶй…Қеӣ°йҡҫвҖ”вҖ”жўҜеәҰйңҖжөҒз»ҸжЁӘи·ЁеӨҡеӣһеҗҲзҡ„ж•°зҷҫдёӘеҫӘзҺҜжӯҘйӘӨгҖӮ

дә”гҖҒеёёи§Ғй—®йўҳ#

Options жЎҶжһ¶дёәдҪ•иғҪеҠ йҖҹеӯҰд№ пјҹ
дёүйҮҚж•Ҳеә”еҸ еҠ пјҡ(i) жңүж•Ҳж—¶еәҸй•ҝеәҰд»Һ $T$ зј©иҮізәҰ $T/\bar k$ пјӣ(ii) й«ҳеұӮеҲҶж”Ҝеӣ еӯҗ $|\mathcal{O}|$ йҖҡеёёиҝңе°ҸдәҺ $|\mathcal{A}|$ пјӣ(iii) intra-option еӯҰд№ дҪҝжҜҸжқЎеҺҹе§ӢиҪ¬з§»йғҪиғҪиҙЎзҢ®з»ҷжүҖжңүе…је®№йҖүйЎ№зҡ„д»·еҖјпјҢиҖҢйқһд»…еҪ“еүҚжҺ§еҲ¶зҡ„йӮЈдёӘгҖӮ

еұӮж¬ЎеҢ–жңҖдјҳдёҺе…ЁеұҖжңҖдјҳжңүдҪ•еҢәеҲ«пјҹ
еұӮж¬ЎеҢ–жңҖдјҳзӯ–з•ҘжҳҜеңЁз»ҷе®ҡеҲҶи§ЈпјҲйҖүйЎ№йӣҶжҲ–д»»еҠЎеӣҫпјүдёӢзҡ„жңҖдјҳи§Јпјӣе…ЁеұҖжңҖдјҳзӯ–з•ҘжҳҜеә•еұӮжүҒе№і MDP дёҠзҡ„жңҖдјҳи§ЈгҖӮеҸӘиҰҒеҲҶи§Јж— жі•иЎЁиҫҫзңҹжӯЈжңҖдјҳзӯ–з•ҘпјҢдәҢиҖ…е°ұдјҡеҒҸзҰ»вҖ”вҖ”дҫӢеҰӮпјҢдёҖдёӘиҮіе°‘иҝҗиЎҢ 10 жӯҘзҡ„йҖүйЎ№ж— жі•е®һзҺ°жҜҸ 3 жӯҘеҲҮжҚўиЎҢдёәзҡ„зӯ–з•ҘгҖӮ

MAML дёәдҪ•йңҖиҰҒдәҢйҳ¶жўҜеәҰпјҹFOMAML жҚҹеӨұжңүеӨҡеӨ§пјҹ
е…ғжҚҹеӨұдёә $\mathcal{L}_{\mathcal{T}_i}(\theta_i')$ пјҢе…¶дёӯ $\theta_i' = \theta - \alpha \nabla_\theta \mathcal{L}_{\mathcal{T}_i}(\theta)$ гҖӮеҜ№ $\theta$ жұӮеҜјеҫ— $(I - \alpha \nabla^2_\theta \mathcal{L}_{\mathcal{T}_i})\, \nabla_{\theta_i'} \mathcal{L}_{\mathcal{T}_i}(\theta_i')$ пјҢеҢ…еҗ« HessianгҖӮFOMAML еҝҪз•Ҙ $-\alpha \nabla^2 \mathcal{L}$ йЎ№пјҢе°Ҷ $\theta_i'$ и§Ҷдёә stop-gradientгҖӮеҺҹи®әж–ҮжҠҘе‘ҠпјҢеңЁж ҮеҮҶе°Ҹж ·жң¬еҹәеҮҶдёҠжҖ§иғҪжҚҹеӨұ <5%гҖӮ

дҪ•ж—¶з”Ё MAMLпјҢдҪ•ж—¶з”Ё RLВІпјҹ
