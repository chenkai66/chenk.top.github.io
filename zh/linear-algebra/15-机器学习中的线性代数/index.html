еӣҫдёӯеұ•зӨәдәҶдёҖдёӘжңүзӣёе…іжҖ§зҡ„дәҢз»ҙзӮ№дә‘пјҢдёӨжқЎдё»иҪҙзҡ„й•ҝеәҰжҢү $\sqrt{\text{ж–№е·®}}$ зј©ж”ҫпјҢдёӢж–№жҳҜж•°жҚ®жҠ•еҪұеҲ° PC1 еҗҺзҡ„дёҖз»ҙеқҗж ҮгҖӮж©ҷиүІзҡ„ PC1 и§ЈйҮҠдәҶеӨ§зәҰ 95% зҡ„зҰ»ж•ЈеәҰпјҢдёўжҺү PC2 еҮ д№ҺдёҚдјҡжҚҹеӨұз»“жһ„дҝЎжҒҜгҖӮ

жҺЁеҜј#

$$\text{Var}(\mathbf{u}) = \frac{1}{n}\sum_{i=1}^n (\mathbf{u}^\top \mathbf{x}_i)^2 = \mathbf{u}^\top \mathbf{C} \mathbf{u}, \qquad \mathbf{C} = \frac{1}{n}\mathbf{X}^\top\mathbf{X}.$$

з”ЁжӢүж јжң—ж—Ҙд№ҳеӯҗжі•жұӮи§Ј $\max \mathbf{u}^\top\mathbf{C}\mathbf{u}$ пјҢзәҰжқҹжқЎд»¶жҳҜ $\|\mathbf{u}\|=1$ пјҢз»“жһңеҫ—еҲ° $\mathbf{C}\mathbf{u} = \lambda\mathbf{u}$ гҖӮжңҖдјҳзҡ„ $\mathbf{u}$ е°ұжҳҜеҚҸж–№е·®зҹ©йҳөжңҖеӨ§зү№еҫҒеҖјеҜ№еә”зҡ„зү№еҫҒеҗ‘йҮҸпјҢеҜ№еә”ж–№еҗ‘зҡ„ж–№е·®е°ұжҳҜ $\lambda$ гҖӮеҗҺз»ӯдё»жҲҗеҲҶжҳҜжҺҘдёӢжқҘзҡ„зү№еҫҒеҗ‘йҮҸпјҢеҪјжӯӨжӯЈдәӨпјҢеӣ дёә $\mathbf{C}$ жҳҜеҜ№з§°зҹ©йҳөгҖӮ

import numpy as np
from sklearn.decomposition import PCA

np.random.seed(42)
X = np.random.randn(200, 2) @ np.array([[2, 0], [0, 0.5]])
theta = np.pi / 6
R = np.array([[np.cos(theta), -np.sin(theta)],
              [np.sin(theta),  np.cos(theta)]])
X = X @ R

pca = PCA(n_components=2).fit(X)
print(pca.components_)               # дё»жҲҗеҲҶж–№еҗ‘пјҲжҜҸиЎҢдёҖдёӘпјү
print(pca.explained_variance_ratio_) # жҜҸдёӘдё»жҲҗеҲҶеҚ жҖ»ж–№е·®зҡ„жҜ”дҫӢ

Xc = X - X.mean(axis=0)
U, S, Vt = np.linalg.svd(Xc, full_matrices=False)
print(Vt)              # дёҺ components_ д»…е·®жӯЈиҙҹеҸ·
print(S**2 / len(X))   # еҗҢж ·зҡ„ж–№е·®

ж”ҜжҢҒеҗ‘йҮҸжңәдёҺж ёжҠҖе·§#

дёәд»Җд№ҲйҖүжӢ©вҖңжңҖеӨ§й—ҙйҡ”вҖқпјҹ#

еҰӮжһңдёӨзұ»ж•°жҚ®жҳҜзәҝжҖ§еҸҜеҲҶзҡ„пјҢйӮЈд№ҲеҸҜд»ҘжүҫеҲ°ж— ж•°дёӘи¶…е№ійқўе°Ҷе®ғд»¬еҲҶејҖгҖӮSVM зҡ„зӣ®ж ҮжҳҜжүҫеҲ°дёҖдёӘе”ҜдёҖзҡ„и¶…е№ійқўпјҢе®ғзҡ„й—ҙйҡ”вҖ”вҖ”еҚіеҲ°дёӨдҫ§жңҖиҝ‘и®ӯз»ғзӮ№зҡ„и·қзҰ»вҖ”вҖ”жҳҜжңҖеӨ§зҡ„гҖӮзӣҙи§ӮдёҠи®ІпјҢй—ҙйҡ”и¶ҠеӨ§пјҢж–°жөӢиҜ•зӮ№иҗҪеңЁжӯЈзЎ®дёҖдҫ§зҡ„еҸҜиғҪжҖ§е°ұи¶Ҡй«ҳпјҢжЁЎеһӢзҡ„жіӣеҢ–иғҪеҠӣд№ҹи¶ҠејәгҖӮиҝҷз§Қзӣҙи§үеҸҜд»ҘйҖҡиҝҮ VC з»ҙе’Ң PAC-Bayes з•ҢзӯүзҗҶи®әдёҘж јиҜҒжҳҺгҖӮ

еӣҫдёӯе®һзәҝжҳҜеҶізӯ–иҫ№з•ҢпјҢж©ҷиүІиҷҡзәҝиЎЁзӨәй—ҙйҡ”иҫ№з•ҢпјҢз»ҝиүІеңҶеңҲж ҮжіЁзҡ„жҳҜжҒ°еҘҪиҗҪеңЁй—ҙйҡ”дёҠзҡ„зӮ№пјҢиҝҷдәӣзӮ№е°ұжҳҜж”ҜжҢҒеҗ‘йҮҸгҖӮеҸӘжңүж”ҜжҢҒеҗ‘йҮҸеҶіе®ҡдәҶжңҖз»Ҳзҡ„и§ЈпјҢеҲ йҷӨд»»дҪ•йқһж”ҜжҢҒеҗ‘йҮҸйғҪдёҚдјҡж”№еҸҳеҶізӯ–иҫ№з•Ңзҡ„дҪҚзҪ®гҖӮ

еҜ№еҒ¶й—®йўҳпјҡж ёеҮҪж•°зҡ„е…ій”®дҪңз”Ё#

еҺҹй—®йўҳпјҡ
$\min_{\mathbf{w}, b} \tfrac{1}{2}\|\mathbf{w}\|^2$ пјҢзәҰжқҹжқЎд»¶дёә $y_i(\mathbf{w}^\top\mathbf{x}_i + b) \ge 1$ гҖӮ

$$\max_{\boldsymbol{\alpha}} \sum_i \alpha_i - \frac{1}{2}\sum_{i,j} \alpha_i \alpha_j y_i y_j \,\mathbf{x}_i^\top \mathbf{x}_j, \qquad \sum_i \alpha_i y_i = 0.$$

еңЁеҜ№еҒ¶й—®йўҳдёӯпјҢж•°жҚ®д»…йҖҡиҝҮеҶ…з§Ҝ $\mathbf{x}_i^\top\mathbf{x}_j$ еҸӮдёҺи®Ўз®—гҖӮеҰӮжһңз”Ёд»»ж„ҸеҮҪж•° $k(\mathbf{x}_i, \mathbf{x}_j)$ жӣҝжҚўиҝҷдәӣеҶ…з§ҜпјҢе°ұзӣёеҪ“дәҺйҡҗејҸең°е°Ҷиҫ“е…Ҙз©әй—ҙжҳ е°„еҲ°еҸҰдёҖдёӘпјҲеҸҜиғҪжӣҙй«ҳз»ҙзҡ„пјүзү№еҫҒз©әй—ҙгҖӮиҝҷе°ұжҳҜж ёжҠҖе·§зҡ„ж ёеҝғжҖқжғігҖӮ

еҚҮз»ҙи§ЈеҶіеӨҚжқӮй—®йўҳ#

еҒҮи®ҫдёӨзұ»ж•°жҚ®еңЁдәҢз»ҙе№ійқўдёҠжҳҜеҗҢеҝғеңҶпјҢжҳҫз„¶ж— жі•з”Ёзӣҙзәҝе°Ҷе®ғд»¬еҲҶејҖгҖӮдҪҶеҰӮжһңйҖҡиҝҮжҳ е°„ $\phi(x_1, x_2) = (x_1, x_2, x_1^2 + x_2^2)$ е°ҶжҜҸдёӘзӮ№жҠ¬еҚҮеҲ°дёүз»ҙз©әй—ҙпјҢеҶ…еңҲзӮ№зҡ„ж–° $z$ еқҗж Үиҫғе°ҸпјҢеӨ–еңҲзӮ№иҫғеӨ§пјҢжӯӨж—¶дёҖдёӘж°ҙе№ійқўе°ұиғҪиҪ»жқҫе°Ҷе®ғд»¬еҲҶејҖгҖӮ

йҖҡеёёжғ…еҶөдёӢпјҢжҳ е°„еҗҺзҡ„з©әй—ҙз»ҙеәҰеҸҜиғҪйқһеёёй«ҳпјҢдҪҶжңүдәҶж ёеҮҪж•°пјҢжҲ‘ж— йңҖжҳҫејҸиЎЁзӨәиҝҷдёӘй«ҳз»ҙз©әй—ҙпјҢеҸӘйңҖи®Ўз®—еҶ…з§ҜеҚіеҸҜгҖӮ

еёёз”Ёж ёеҮҪж•°пјҡ

  • зәҝжҖ§ж ёпјҡ $k(\mathbf{x}, \mathbf{y}) = \mathbf{x}^\top\mathbf{y}$
  • $d$ йҳ¶еӨҡйЎ№ејҸж ёпјҡ $k(\mathbf{x}, \mathbf{y}) = (\mathbf{x}^\top\mathbf{y} + c)^d$
  • RBF пјҲй«ҳж–Ҝпјүж ёпјҡ $k(\mathbf{x}, \mathbf{y}) = \exp(-\gamma\|\mathbf{x} - \mathbf{y}\|^2)$ вҖ”вҖ”еҜ№еә”ж— йҷҗз»ҙзү№еҫҒз©әй—ҙпјҢдҪҶи®Ўз®—еӨҚжқӮеәҰд»…дёә $O(p)$ гҖӮ
  • Sigmoid ж ёпјҡ $k(\mathbf{x}, \mathbf{y}) = \tanh(\kappa \mathbf{x}^\top\mathbf{y} + c)$ вҖ”вҖ”е№¶йқһжҖ»жҳҜжӯЈе®ҡпјҢеҺҶеҸІдёҠжӣҫз”ЁдәҺвҖңзҘһз»ҸеҢ–вҖқSVMгҖӮ

з”ЁжҲ·-з”өеҪұиҜ„еҲҶзҹ©йҳө $\mathbf{R} \in \mathbb{R}^{m\times n}$ йқһеёёеәһеӨ§пјҢеҢ…еҗ«ж•°еҚғдёҮз”ЁжҲ·е’Ңж•°еҚҒдёҮйғЁз”өеҪұпјҢдҪҶж•°жҚ®еҮ д№Һе…ЁжҳҜз©әзҡ„вҖ”вҖ”жҷ®йҖҡз”ЁжҲ·еҸӘдјҡз»ҷзӣ®еҪ•дёӯ 0.1% зҡ„еҶ…е®№жү“еҲҶгҖӮжҲ‘зҡ„зӣ®ж ҮжҳҜйў„жөӢйӮЈдәӣз©әзҷҪж јеӯҗзҡ„еҲҶж•°пјҢз„¶еҗҺжҠҠйў„жөӢеҫ—еҲҶжңҖй«ҳзҡ„жңӘи§ӮзңӢеҪұзүҮжҺЁиҚҗз»ҷжҜҸдёӘз”ЁжҲ·гҖӮNetflix Prize пјҲ2006-2009пјүеҘ е®ҡдәҶзҹ©йҳөеҲҶи§Јзҡ„ж ёеҝғең°дҪҚпјҢеҰӮд»Ҡ SpotifyгҖҒYouTube е’Ңеҗ„еӨ§з”өе•Ҷе№іеҸ°зҡ„жҺЁиҚҗзі»з»ҹдҫқз„¶еңЁз”Ёе®ғзҡ„еҸҳз§ҚгҖӮ

дҪҺз§©еҒҮи®ҫ#

$$\hat{r}_{uj} = \mathbf{p}_u^\top \mathbf{q}_j, \qquad \mathbf{R} \approx \mathbf{P}\mathbf{Q}^\top, \quad \mathbf{P} \in \mathbb{R}^{m\times k}, \mathbf{Q} \in \mathbb{R}^{n\times k}.$$

иҝҷе…¶е®һе°ұжҳҜеҜ№ $\mathbf{R}$ еҒҡз§©-$k$ иҝ‘дјјгҖӮеҪ“ $m=10^7, n=10^5, k=64$ ж—¶пјҢеӣ еӯҗзҹ©йҳөеҸӘйңҖиҰҒеӯҳеӮЁеӨ§зәҰ $6\times 10^8$ дёӘж•°еӯ—пјҢиҖҢдёҚжҳҜ $10^{12}$ дёӘгҖӮиҖҢдё”иҝҷз§Қж–№жі•иҝҳиғҪжҺЁе№ҝеҲ°жңӘи§ӮжөӢзҡ„ж•°жҚ®дёҠгҖӮ

еӣҫдёӯеұ•зӨәдәҶзЁҖз–Ҹзҡ„и§ӮжөӢзҹ©йҳөпјҲ? иЎЁзӨәжңӘиҜ„еҲҶпјүгҖҒдёӨдёӘеӣ еӯҗзҹ©йҳөпјҢд»ҘеҸҠиҝҳеҺҹеҮәзҡ„зЁ еҜҶйў„жөӢзҹ©йҳөгҖӮ

дәӨжӣҝжңҖе°ҸдәҢд№ҳжі•пјҲALSпјү#

$$\min_{\mathbf{P}, \mathbf{Q}} \sum_{(u,j)\in\Omega}\bigl(r_{uj} - \mathbf{p}_u^\top\mathbf{q}_j\bigr)^2 + \lambda\bigl(\|\mathbf{P}\|_F^2 + \|\mathbf{Q}\|_F^2\bigr).$$

иҝҷдёӘзӣ®ж ҮеҮҪж•°еҜ№ $(\mathbf{P}, \mathbf{Q})$ иҒ”еҗҲжҳҜйқһеҮёзҡ„пјҢдҪҶеҰӮжһңеӣәе®ҡдёҖдёӘпјҢеҸҰдёҖдёӘе°ұжҳҜеҮёзҡ„гҖӮALS е°ұжҳҜиҝҷж ·дәӨжӣҝиҝӣиЎҢзҡ„пјҡеӣәе®ҡ $\mathbf{Q}$ ж—¶пјҢ$\mathbf{P}$ зҡ„жҜҸдёҖиЎҢйғҪжҳҜдёҖдёӘзӢ¬з«Ӣзҡ„еІӯеӣһеҪ’й—ӯејҸи§ЈпјӣеҸҚиҝҮжқҘд№ҹдёҖж ·гҖӮжҜҸиҪ®иҝӯд»Јдјҡжү«дёҖйҒҚжүҖжңүе·ІзҹҘиҜ„еҲҶпјҢжҜҸиЎҢжұӮи§ЈеҸҜд»ҘиҪ»жқҫе№¶иЎҢвҖ”вҖ”иҝҷе°ұжҳҜ ALS еңЁ Spark MLlib дёӯеҚ жҚ®дё»еҜјең°дҪҚзҡ„еҺҹеӣ гҖӮ

import numpy as np

def als(R, mask, k=10, n_iter=20, lam=0.1):
    m, n = R.shape
    P = np.random.randn(m, k) * 0.1
    Q = np.random.randn(n, k) * 0.1
    I = lam * np.eye(k)
    for _ in range(n_iter):
        for u in range(m):
            j = np.where(mask[u])[0]
            if j.size:
                Qj = Q[j]
                P[u] = np.linalg.solve(Qj.T @ Qj + I, Qj.T @ R[u, j])
        for j in range(n):
            u = np.where(mask[:, j])[0]
            if u.size:
                Pu = P[u]
                Q[j] = np.linalg.solve(Pu.T @ Pu + I, Pu.T @ R[u, j])
    return P, Q

  • еҠ е…Ҙиҫ…еҠ©дҝЎжҒҜпјҢжҜ”еҰӮдәәеҸЈз»ҹи®Ўзү№еҫҒжҲ–еҶ…е®№е…ғж•°жҚ®пјҢжһ„е»әж··еҗҲжЁЎеһӢпјҡ$\hat{r}_{uj} = \mathbf{p}_u^\top\mathbf{q}_j + \mathbf{a}^\top \mathbf{f}_u + \mathbf{b}^\top \mathbf{g}_j$ гҖӮ
  • еј•е…Ҙз”ЁжҲ·е’Ңзү©е“ҒеҒҸзҪ®пјҡжңүдәӣз”ЁжҲ·жү“еҲҶжҷ®йҒҚеҒҸй«ҳпјҢжңүдәӣз”өеҪұжң¬иә«жӣҙеҸ—ж¬ўиҝҺгҖӮз”Ёе…¬ејҸ $\hat{r}_{uj} = \mu + b_u + b_j + \mathbf{p}_u^\top\mathbf{q}_j$ еҸҜд»Ҙжҳҫи‘—жҸҗеҚҮйў„жөӢзІҫеәҰгҖӮ

жЁЎеһӢ#

$$\mathbf{y} = \mathbf{X}\boldsymbol{\beta} + \boldsymbol{\epsilon}, \qquad \mathbf{X} \in \mathbb{R}^{n\times(p+1)}.$$

е…¶дёӯ $\mathbf{X}$ жҳҜи®ҫи®Ўзҹ©йҳөпјҢз¬¬дёҖеҲ—е…ЁжҳҜ 1 пјҲеҜ№еә”жҲӘи·қйЎ№пјүгҖӮ

жңҖе°ҸдәҢд№ҳзҡ„жң¬иҙЁжҳҜжӯЈдәӨжҠ•еҪұ#

зӣ®ж ҮжҳҜжңҖе°ҸеҢ–ж®Ӣе·®е№іж–№е’Ң $\|\mathbf{y} - \mathbf{X}\boldsymbol{\beta}\|^2$ гҖӮд»ҺеҮ дҪ•дёҠзңӢпјҢ$\mathbf{X}\boldsymbol{\beta}$ зҡ„еҸ–еҖјиҢғеӣҙжҳҜ $\mathbf{X}$ зҡ„еҲ—з©әй—ҙгҖӮжҲ‘иҰҒжүҫзҡ„жҳҜиҝҷдёӘеӯҗз©әй—ҙйҮҢзҰ» $\mathbf{y}$ жңҖиҝ‘зҡ„зӮ№гҖӮжңҖиҝ‘зҡ„зӮ№е°ұжҳҜеһӮи¶іпјҢд№ҹе°ұжҳҜ $\mathbf{y}$ еңЁеҲ—з©әй—ҙдёҠзҡ„жӯЈдәӨжҠ•еҪұгҖӮ

$$\hat{\boldsymbol{\beta}} = (\mathbf{X}^\top\mathbf{X})^{-1}\mathbf{X}^\top\mathbf{y}.$$

е®һйҷ…еҶҷд»Јз Ғж—¶пјҢжҲ‘дёҚдјҡзӣҙжҺҘжұӮйҖҶзҹ©йҳөпјҢиҖҢжҳҜз”Ё np.linalg.lstsq(X, y)пјҲеә•еұӮз”ЁзЁіе®ҡзҡ„ QR жҲ– SVD жұӮи§ЈеҷЁпјүпјҢжҲ–иҖ…жҳҫејҸеҲҶи§Ј $\mathbf{X} = \mathbf{Q}\mathbf{R}$ еҗҺеӣһд»ЈжұӮи§Ј $\mathbf{R}\hat{\boldsymbol{\beta}} = \mathbf{Q}^\top\mathbf{y}$ гҖӮ

еІӯеӣһеҪ’пјҡйҖҡиҝҮж”¶зј©ж”№е–„жқЎд»¶ж•°#

$$\hat{\boldsymbol{\beta}}_{\text{ridge}} = (\mathbf{X}^\top\mathbf{X} + \lambda\mathbf{I})^{-1}\mathbf{X}^\top\mathbf{y}.$$

еҠ дёҠ $\lambda\mathbf{I}$ еҗҺпјҢ$\mathbf{X}^\top\mathbf{X}$ зҡ„жҜҸдёӘзү№еҫҒеҖјйғҪеўһеҠ дәҶ $\lambda$ пјҢд»ҺиҖҢдҝқиҜҒзҹ©йҳөеҸҜйҖҶпјҢе№¶е°ҶжқЎд»¶ж•°д»Һ $\sigma_{\max}^2/\sigma_{\min}^2$ зј©е°ҸеҲ°еӨ§зәҰ $(\sigma_{\max}^2 + \lambda)/(\sigma_{\min}^2 + \lambda)$ гҖӮд»Һ SVD зҡ„и§’еәҰзңӢпјҢеІӯеӣһеҪ’жҠҠ OLS и§Јзҡ„жҜҸдёӘеҘҮејӮеҲҶйҮҸд№ҳд»Ҙ $\sigma_i^2/(\sigma_i^2 + \lambda)$ вҖ”вҖ”е°ҸеҘҮејӮеҖјпјҲеҷӘеЈ°ж–№еҗ‘пјүиў«еӨ§е№…еҺӢзј©пјҢиҖҢеӨ§еҘҮејӮеҖјпјҲдҝЎеҸ·ж–№еҗ‘пјүеҮ д№ҺдёҚеҸҳгҖӮ

LASSOпјҡз”Ё $\ell_1$ е®һзҺ°зЁҖз–ҸжҖ§#

$$\min_{\boldsymbol{\beta}} \|\mathbf{y} - \mathbf{X}\boldsymbol{\beta}\|^2 + \lambda \|\boldsymbol{\beta}\|_1.$$

$\ell_1$ иҢғж•°зҡ„зӯүеҖјйқўжҳҜдёҖдёӘиҸұеҪўпјҢи§’зӮ№жӯЈеҘҪиҗҪеңЁеқҗж ҮиҪҙдёҠгҖӮOLS зҡ„зӯүй«ҳзәҝйҖҡеёёдјҡеңЁжҹҗдёӘи§’зӮ№дёҺ $\ell_1$ иҢғж•°зӣёеҲҮпјҢиҝҷж—¶жҹҗдәӣзі»ж•°жҒ°еҘҪдёәйӣ¶гҖӮжұӮи§ЈеҷЁдјҡиҮӘеҠЁе®ҢжҲҗзү№еҫҒйҖүжӢ©вҖ”вҖ”иҝҷеҜ№зЁҖз–Ҹй«ҳз»ҙй—®йўҳпјҲеҰӮеҹәеӣ з»„еӯҰгҖҒNLPгҖҒй«ҳйў‘дәӨжҳ“дҝЎеҸ·пјүжқҘиҜҙйқһеёёе®һз”ЁгҖӮ

$$\mathbf{h} = \sigma(\mathbf{W}\mathbf{x} + \mathbf{b}).$$

еҰӮжһңжІЎжңү $\sigma$ пјҢеӨҡеұӮеҸ еҠ е°ұдјҡйҖҖеҢ–жҲҗеҚ•еұӮпјҡ$\mathbf{W}_3\mathbf{W}_2\mathbf{W}_1$ дёҚиҝҮжҳҜеҸҰдёҖдёӘзҹ©йҳөгҖӮжҝҖжҙ»еҮҪж•°жү“з ҙдәҶзәҝжҖ§йҷҗеҲ¶пјҢи®©зҪ‘з»ңиғҪеӨҹйҖјиҝ‘д»»ж„Ҹиҝһз»ӯеҮҪж•°пјҲдёҮиғҪйҖјиҝ‘е®ҡзҗҶпјүгҖӮ

жү№йҮҸеӨ„зҗҶпјҡд»Һзҹ©йҳө-еҗ‘йҮҸд№ҳеҲ°зҹ©йҳө-зҹ©йҳөд№ҳ#

$$\mathbf{H} = \sigma\bigl(\mathbf{X}\mathbf{W}^\top + \mathbf{1}\mathbf{b}^\top\bigr).$$

дёҖж¬Ў SGEMM и°ғз”Ёе°ұиғҪе®ҢжҲҗеҺҹжң¬йңҖиҰҒ $B$ ж¬ЎзӢ¬з«Ӣзҹ©йҳө-еҗ‘йҮҸд№ҳжі•зҡ„е·ҘдҪңпјҢFLOP еҲ©з”ЁзҺҮеӨ§е№…жҸҗеҚҮгҖӮ

import numpy as np

class LinearLayer:
    """жүӢеҶҷдёҖдёӘзҹ©йҳөд№ҳжі•еұӮпјҢеүҚеҗ‘е’ҢеҸҚеҗ‘йғҪиҮӘе·ұе®һзҺ°"""
    def __init__(self, d_in, d_out):
        self.W = np.random.randn(d_out, d_in) * np.sqrt(2.0 / d_in)
        self.b = np.zeros(d_out)

    def forward(self, x):
        self.x = x
        return x @ self.W.T + self.b

    def backward(self, grad_out, lr=1e-2):
        grad_W = grad_out.T @ self.x
        grad_b = grad_out.sum(axis=0)
        grad_x = grad_out @ self.W
        self.W -= lr * grad_W
        self.b -= lr * grad_b
        return grad_x

жіЁж„ҸеҠӣжңәеҲ¶д№ҹжҳҜзәҝжҖ§д»Јж•°#

$$\text{Attention}(\mathbf{Q}, \mathbf{K}, \mathbf{V}) = \text{softmax}\!\left(\frac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d_k}}\right)\mathbf{V},$$

е…¶дёӯ $\mathbf{Q} = \mathbf{X}\mathbf{W}_Q$ пјҢ$\mathbf{K} = \mathbf{X}\mathbf{W}_K$ пјҢ$\mathbf{V} = \mathbf{X}\mathbf{W}_V$ жҳҜеҜ№еҗҢдёҖиҫ“е…Ҙзҡ„дёүз§ҚзәҝжҖ§жҠ•еҪұгҖӮ$\mathbf{Q}\mathbf{K}^\top$ жҳҜдёҖдёӘ $n\times n$ зҡ„зӣёдјјеәҰзҹ©йҳөпјҢиЎЁзӨәеәҸеҲ—дёӯжҜҸеҜ№дҪҚзҪ®д№Ӣй—ҙзҡ„зӣёдјјжҖ§пјӣsoftmax е°Ҷе…¶иҪ¬еҢ–дёәжҰӮзҺҮзҹ©йҳөпјӣеҶҚд№ҳд»Ҙ $\mathbf{V}$ е°ұжҳҜеҜ№еҖјеҒҡеҠ жқғе№іеқҮгҖӮиҝҷдёӘзӣёдјјеәҰзҹ©йҳөзҡ„ $O(n^2)$ еҶ…еӯҳејҖй”ҖпјҢжӯЈжҳҜ FlashAttention е’ҢзәҝжҖ§жіЁж„ҸеҠӣз ”з©¶иҜ•еӣҫи§ЈеҶізҡ„й—®йўҳгҖӮ

дјҳеҢ–з®—жі•зҡ„зәҝжҖ§д»Јж•°еҹәзЎҖ#

жўҜеәҰдёҺ Hessian#

$$f(\mathbf{x}) \approx f(\mathbf{x}_0) + \nabla f^\top(\mathbf{x} - \mathbf{x}_0) + \tfrac{1}{2}(\mathbf{x} - \mathbf{x}_0)^\top \mathbf{H}(\mathbf{x} - \mathbf{x}_0).$$

еҰӮжһң Hessian жҳҜеҜ№з§°жӯЈе®ҡзҹ©йҳөпјҢиҜҙжҳҺиҝҷйҮҢжҳҜдёҖдёӘзңҹжӯЈзҡ„жһҒе°ҸеҖјзӮ№гҖӮHessian зҡ„зү№еҫҒеҖјеҜ№еә”дәҶе…¶зү№еҫҒж–№еҗ‘дёҠзҡ„жӣІзҺҮгҖӮ

жқЎд»¶ж•°дёҺвҖңд№ӢвҖқеӯ—еҪўзҺ°иұЎ#

жўҜеәҰдёӢйҷҚзҡ„ж”¶ж•ӣйҖҹеәҰеҸ–еҶідәҺ Hessian зҡ„жқЎд»¶ж•° $\kappa = \lambda_{\max}/\lambda_{\min}$ гҖӮеҪ“жқЎд»¶ж•°еҫҲеӨ§ж—¶пјҢзӣ®ж ҮеҮҪж•° $f$ зҡ„зӯүй«ҳзәҝдјҡеҸҳжҲҗй«ҳеәҰжӢүй•ҝзҡ„жӨӯеңҶгҖӮжӯӨж—¶жўҜеәҰж–№еҗ‘жЁӘеҲҮжӨӯеңҶпјҢиҖҢдёҚжҳҜжІҝзқҖжӨӯеңҶдёӢйҷҚпјҢеҜјиҮҙи‘—еҗҚзҡ„вҖңд№ӢвҖқеӯ—еҪўзҺ°иұЎгҖӮеҜ№дәҺдәҢж¬Ўзӣ®ж ҮеҮҪж•°пјҢжҷ®йҖҡжўҜеәҰдёӢйҷҚжҜҸдёҖжӯҘзҡ„ж”¶ж•ӣзҺҮдёә $\bigl((\kappa-1)/(\kappa+1)\bigr)^2$ гҖӮжҜ”еҰӮеҪ“ $\kappa = 1000$ ж—¶пјҢж”¶ж•ӣзҺҮзәҰдёә 0.996пјҢеҮ д№ҺжІЎд»Җд№Ҳиҝӣеұ•гҖӮ

Adam еҲ°еә•еңЁеҒҡд»Җд№Ҳ#

зүӣйЎҝжі•йҖҡиҝҮ $\mathbf{H}^{-1}$ еҜ№й—®йўҳиҝӣиЎҢйў„еӨ„зҗҶпјҢе°ҶеҮ дҪ•еҪўзҠ¶еҸҳжҚўдёә Hessian жҲҗдёәеҚ•дҪҚзҹ©йҳөзҡ„еҪўејҸпјҲеҜ№дәҺдәҢж¬Ўй—®йўҳеҸҜд»ҘдёҖжӯҘеҲ°дҪҚпјүгҖӮдҪҶи®Ўз®—еҚҒдәҝеҸӮж•°зҪ‘з»ңзҡ„ $\mathbf{H}^{-1}$ ж №жң¬дёҚзҺ°е®һпјҢжүҖд»Ҙ Adam дҪҝз”ЁдәҶдёҖдёӘеҜ№и§’йў„еӨ„зҗҶеҷЁжқҘиҝ‘дјје®ғвҖ”вҖ”е…·дҪ“жқҘиҜҙпјҢе°ұжҳҜеҜ№жҜҸдёӘеҸӮж•°з»ҙжҠӨдёҖдёӘжўҜеәҰе№іж–№зҡ„ж»‘еҠЁе№іеқҮеҖј $\hat{v}_t$ гҖӮжӣҙж–°е…¬ејҸ $\theta \leftarrow \theta - \eta \,\hat{m}_t / \sqrt{\hat{v}_t}$ еҜ№жҜҸдёӘеқҗж ҮйҮҚж–°зј©ж”ҫпјҡй«ҳжӣІзҺҮж–№еҗ‘жӯҘе№…е°ҸпјҢдҪҺжӣІзҺҮж–№еҗ‘жӯҘе№…еӨ§пјҢд»ҺиҖҢд»Ҙ $O(p)$ зҡ„йўқеӨ–ејҖй”Җи§ЈеҶідәҶвҖңд№ӢвҖқеӯ—еҪўй—®йўҳгҖӮ
