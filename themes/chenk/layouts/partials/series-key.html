{{- /* Normalize a page's series value to a lowercase, dash-joined key. Series may be a string OR a map with .slug/.key/.name/.part/.total. We try map fields first; fall back to string conversion. */ -}} {{- $s := . -}} {{- $key := "" -}} {{- if $s -}} {{- $t := printf "%T" $s -}} {{- if eq $t "string" -}} {{- $key = $s -}} {{- else -}} {{- /* assume map-ish */ -}} {{- $key = (index $s "slug") -}} {{- if not $key -}}{{- $key = (index $s "key") -}}{{- end -}} {{- if not $key -}}{{- $key = (index $s "name") -}}{{- end -}} {{- if not $key -}}{{- $key = printf "%v" $s -}}{{- end -}} {{- end -}} {{- end -}} {{- if not $key -}}{{- $key = "standalone" -}}{{- end -}} {{- $key = lower (printf "%v" $key) -}} {{- $key = replace $key " " "-" -}} {{- $key -}}
